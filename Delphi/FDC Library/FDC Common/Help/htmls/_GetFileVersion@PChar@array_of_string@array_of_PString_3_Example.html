GetFileVersion
  var
    InternalName, FileVersion: string;
  begin
    GetFileVersion(PChar(Application.ExeName),
      ['InternalName', 'FileVersion'],
      [@InternalName, @FileVersion]);
    Label1.Caption := InternalName + ' ' + FileVersion;
  end;
